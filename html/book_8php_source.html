76 $start = new DateTime($start);
77 $end = new DateTime($end);
78 $interval = new DateInterval("PT".$duration."M");
79 $cleanupInterval = new DateInterval("PT".$cleanup."M");
80 $slots = array();
81
82 for ($intStart = $start; $intStart < $end; $intStart->add($interval)->add($cleanupInterval)) {
83 $endPeriod = clone $intStart;
84 $endPeriod->add($interval);
85 if ($endPeriod > $end) {
86 break;

90 }
